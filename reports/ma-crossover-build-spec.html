Technical Specification

Pine Script Code Structure

//@version=5
indicator("Chase: MA Crossover Alerts", shorttitle="MA Cross Alert", overlay=true)

// === INPUTS ===
ma1_period = input.int(7, title="MA1 Period", minval=1, maxval=200)
ma2_period = input.int(21, title="MA2 Period", minval=1, maxval=200)
ma_type = input.string("EMA", title="MA Type", options=["EMA", "SMA"])
color_bull = input.color(color.new(#00D084, 0), title="Bullish Color")
color_bear = input.color(color.new(#FF0000, 0), title="Bearish Color")
show_alerts = input.bool(true, title="Show Alerts")

// === CALCULATIONS ===
ma1 = ta.ema(close, ma1_period) if ma_type == "EMA" else ta.sma(close, ma1_period)
ma2 = ta.ema(close, ma2_period) if ma_type == "EMA" else ta.sma(close, ma2_period)

// === LOGIC ===
crossover = ta.crossover(ma1, ma2)
crossunder = ta.crossunder(ma1, ma2)

// === PLOTS ===
plot(ma1, color=color.new(color_bull, 30), linewidth=2, title="MA1")
plot(ma2, color=color.new(color_bear, 30), linewidth=2, title="MA2")

// === ALERTS ===
if crossover and show_alerts
    label.new(bar_index, high, "Bullish Cross", yloc=yloc.aboveclose, color=color_bull)
